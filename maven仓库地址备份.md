我们还可以设置以下的Maven仓库地址，如下：

<mirrors>
    <mirror>
      <id>wso2maven</id>
      <name>wso2 maven</name>
      <url>http://maven.wso2.org/nexus/content/groups/public/</url>
      <mirrorof>central</mirrorof>        
    </mirror>
</mirrors>
